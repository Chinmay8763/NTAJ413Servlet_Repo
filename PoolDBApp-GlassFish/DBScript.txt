CREATE TABLE "SCOTT"."CORONA_REGISTRATION" 
   (	"PATID" NUMBER(5,0) NOT NULL ENABLE, 
	"PATNAME" VARCHAR2(15 BYTE), 
	"PATADDRS" VARCHAR2(15 BYTE), 
	"AGE" NUMBER(3,0), 
	"GENDER" VARCHAR2(7 BYTE), 
	"STAGE" VARCHAR2(10 BYTE), 
	 CONSTRAINT "CORONA_REGISTRATION_PK" PRIMARY KEY ("PATID");
	 
	 
   CREATE SEQUENCE  "SCOTT"."CORONA_PATID_SEQUENCE"  MINVALUE 0
